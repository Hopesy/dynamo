模板二，添加了注释

# 加载 Python Standard 和 DesignScript 库
import sys #引入Python标准库,可以用来指定dll文件的路径
sys.path.append('C:\Program Files (x86)\IronPython 2.7\Lib')
import clr #引入clr(Common Language Runtime)库。它是Python程序运行在.net平台的基础。
#如果不是通过Python编写的模块。例如.NET库，需要通过CLR模块来引用。而且将这个文件引用之后，需要从它的命名空间中导入类。
clr.AddReference('ProtoGeometry') #引入Dynamo自身的图形处理库
from Autodesk.DesignScript.Geometry import * #引入库下面的所有类和方法

# 导入 RevitAPI 和 RevitAPIUI
clr.AddReference("RevitAPI") #导入RevitAPI.dll文件，文件后缀.dll可以省略
clr.AddReference("RevitAPIUI") #导入RevitAPIUI.dll文件，文件后缀.dll可以省略

import Autodesk #导入Autodesk命名空间，这个导入命名空间可以省略，最好加上吧
#如果只导入namespace没有下面的from * import *调用的时候前面需要加NAmespace.func()
from Autodesk.Revit.DB import *  #再从具体的命名空间下引入模块，这样可以直接func()
from Autodesk.Revit.UI import *
clr.AddReference("RevitNodes") #导入Dynamo封装库RevitNodes.dll，这样可以在Python Script里面直接写Design Script代码
import Revit #同上Autodesk
from Revit.Elements import * #导入之后在Python Script调用Dyna节点函数和电池一样

#导入List对应C#中的Icollection
clr.AddReference("System") #导入.net中的system
from System.Collections.Generic import List
