# 进度条

进度条,定义进度条的类如下。我们只需要把它引入我们的程序中实例化即可。

种树实例

import sys
import clr
import System
clr.AddReference("RevitServices")
clr.AddReference("RevitNodes")
import RevitServices
from RevitServices.Persistence import DocumentManager
from RevitServices.Transactions import TransactionManager
doc =  DocumentManager.Instance.CurrentDBDocument
import Revit
clr.ImportExtensions(Revit.Elements)
clr.ImportExtensions(Revit.GeometryConversion)
clr.AddReference("RevitAPI")
from Autodesk.Revit.DB import *
famType = UnwrapElement(IN[0])#输入FamilyType
pts = IN[1] #输入种树的点
if not isinstance(pts, list):
  pts = UnwrapElement([pts])
else:
  pts = UnwrapElement(pts)
ProgressBarUtils = IN[2]
pgb = ProgressBarUtils()#实例化进度条类
elems = []
elemlist = []
##len(pts)就是进度条的显示时间，进度条存在的时间是broadcaster:后面程序的执行时间
with pgb.MyProgressBroadcaster(len(pts), "种树进度") as broadcaster:
  TransactionManager.Instance.EnsureInTransaction(doc)
  for pt in pts:
    #你自己的程序操作逻辑
    if famType.IsActive == False:
      famType.Activate()#将族类型激活，不要漏了这一步
    elems = doc.Create.NewFamilyInstance(pt.ToRevitType(), famType, Structure.StructuralType.NonStructural)
    #ToRevitType()将Dynamo的point转化为Revit Api中的点
    elemlist.append(elems.ToDSType(True))#ToDSType将生成的elem所有权给Revit，避免二次运行消失
    #通过EventHook增加进度条
    broadcaster.onChange.next_p()
  TransactionManager.Instance.TransactionTaskDone()
OUT = elemlist
